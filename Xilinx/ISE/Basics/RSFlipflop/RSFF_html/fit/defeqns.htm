FDCPE_Q: FDCPE port map (Q,Q_D,clk,Q_CLR,set);
     Q_D <= ((S)
      OR (Q AND R));
     Q_CLR <= (NOT set AND reset);
FDCPE_Qn: FDCPE port map (Qn,Qn_D,clk,set,reset);
     Qn_D <= ((Qn AND NOT S)
      OR (NOT S AND NOT R));
